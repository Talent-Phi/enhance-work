Opción JS (sin PHP)

Para CAPI en JS puro, el hashing SHA-256 se hace con la Web Crypto API del browser y el fetch va directo a Meta. El único trade-off es que el token queda visible en el JS del browser — para mitigarlo se puede restringir el token solo a send_events en Meta App settings.

Cómo instalarlo en WordPress:

1. Instalar el plugin gratuito "WPCode" (o "Insert Headers and Footers")
2. Ir a Code Snippets → Header
3. Pegar el script de abajo

  function genEventId() {

  // ── SHA-256 con Web Crypto API
  async function sha256(str) {
    if (!str) return null;
    var clean = str.trim().toLowerCase();
    var buf   = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(clean));
    return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2,'0')).join('');
  }

  // ── Enviar evento a Meta CAPI
  async function sendCAPI(eventName, eventId, userData) {
    var userDataHashed = {
      client_user_agent: navigator.userAgent,
      client_ip_address: '' // no disponible en browser
    };

    // fbp / fbc desde cookies
    var cookies = document.cookie.split(';').reduce(function(acc, c) {
      var p = c.trim().split('='); acc[p[0]] = p[1]; return acc;
    }, {});
    if (cookies['_fbp']) userDataHashed['fbp'] = cookies['_fbp'];
    if (cookies['_fbc']) userDataHashed['fbc'] = cookies['_fbc'];

    // Hashear PII
    if (userData.email)      userDataHashed['em'] = await sha256(userData.email);
    if (userData.phone)      userDataHashed['ph'] = await sha256(userData.phone.replace(/\D/g,''));
    if (userData.first_name) userDataHashed['fn'] = await sha256(userData.first_name);
    if (userData.last_name)  userDataHashed['ln'] = await sha256(userData.last_name);
    if (userData.dob)        userDataHashed['db'] = await sha256(userData.dob.replace(/\D/g,''));

    // Limpiar nulls
    Object.keys(userDataHashed).forEach(function(k) {
      if (!userDataHashed[k]) delete userDataHashed[k];
    });

    var payload = {
      data: [{
        event_name:       eventName,
        event_time:       Math.floor(Date.now() / 1000),
        event_id:         eventId,
        event_source_url: window.location.href,
        action_source:    'website',
        user_data:        userDataHashed
      }]
    };
    if (TEST_CODE) payload.test_event_code = TEST_CODE;

      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      body:    JSON.stringify(payload)
